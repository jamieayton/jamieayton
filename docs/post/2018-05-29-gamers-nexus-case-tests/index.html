# combine data
combined_data <- bind_rows(
  cpu_test_data, gpu_test_data
) %>% 
mutate(
  `Test` = str_replace_all(`Test`, "Case Torture – ", ""), 
  `Case & Setup` = paste0(`Case`, " - ", `Setup`)
) %>% 
gather(
  `Type`, `Temperature`, c("Idle", "Load")
) %>% 
spread(
  `Test`, `Temperature`
) %>% 
drop_na() %>% 
filter(
  `Type` == "Load"
)

# add lm fit for plot
temperature_lm <- lm(`GPU Temperature` ~ `CPU Temperature`, data = combined_data)

# add predictors to dataframe
combined_data <- combined_data %>% 
  modelr::add_predictions(model = temperature_lm)

combined_data %>% 
  ggplot(aes(x = `CPU Temperature`, y = `GPU Temperature`, label = `Case`)) + 
  geom_smooth(method = "lm", se = FALSE, linetype = 3) + 
  geom_label_repel(
    size = 3, 
    nudge_x = ifelse(combined_data$`GPU Temperature` > combined_data$pred, 0/4, 10/4),
    nudge_y = ifelse(combined_data$`GPU Temperature` > combined_data$pred, 7/4, 2/4),
    segment.size  = 0.2,
    segment.color = "grey50",
    direction     = "both", 
    box.padding = unit(0.5, "lines"), 
    force = 10
  ) + 
  geom_point(colour = "red", size = 3) + 
  theme_minimal()
